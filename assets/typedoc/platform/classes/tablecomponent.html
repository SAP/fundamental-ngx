The component that represents a table. A table is a set of tabular data. Line items can support data, images and actions.

<fdp-table
 [dataSource]="source"
 fdCompact
 selectionMode="multiple"
 emptyTableMessage="No data found">

 <fdp-table-toolbar
  title="Order Line Items"
  [hideItemCount]="false">
 </fdp-table-toolbar>

 <fdp-column
  name="name"
  key="name"
  label="Name"
  align="start">
 </fdp-column>

 <fdp-column
  name="description"
  key="description"
  label="Description">
 </fdp-column>
</fdp-table>
